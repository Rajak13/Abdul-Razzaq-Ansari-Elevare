# ========================================
# API CONFIGURATION
# ========================================
NEXT_PUBLIC_API_URL=https://<YOUR_RENDER_BACKEND_URL>
NEXT_PUBLIC_WS_URL=wss://<YOUR_RENDER_BACKEND_URL>

# Example:
# NEXT_PUBLIC_API_URL=https://elevare-backend-xxxx.onrender.com
# NEXT_PUBLIC_WS_URL=wss://elevare-backend-xxxx.onrender.com

# ========================================
# HOW TO ADD THESE TO VERCEL:
# ========================================
# 1. Go to your Vercel project dashboard
# 2. Click "Settings" tab
# 3. Click "Environment Variables" in the left sidebar
# 4. For each variable above:
#    - Enter the Key (e.g., NEXT_PUBLIC_API_URL)
#    - Enter the Value (e.g., https://your-backend.onrender.com)
#    - Select "Production" environment
#    - Click "Save"
# 5. After adding all variables, go to "Deployments" tab
# 6. Click the three dots on the latest deployment
# 7. Click "Redeploy" to apply the new environment variables

# ========================================
# FINDING YOUR RENDER BACKEND URL:
# ========================================
# 1. Go to your Render dashboard
# 2. Click on your backend service
# 3. Look at the top of the page - you'll see a URL like:
#    https://elevare-backend-xxxx.onrender.com
# 4. Copy this URL and use it in the variables above
